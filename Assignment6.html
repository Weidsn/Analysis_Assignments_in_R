Import Libraries

library(tidyverse)
library(stats)
library(forecast)

Load the data (already cleaned)

df_retail <- read.csv("retail_data.csv")

Create time series on Sales, and apply log() to the time series

ts_retail <- ts(df_retail$Sales,
                start = c(2018,1),
                end = c(2023,12),
                frequency = 12)
# taking the log()
ts_retail_log <- log(ts_retail)
glimpse(ts_retail_log)
##  Time-Series [1:72] from 2018 to 2024: 9.58 9.25 9.7 9.9 9.04 ...

Plot the decomposition

decompose(ts_retail_log) |> 
  plot()

Apply ARIMA and display summary

arima_retail <- auto.arima(ts_retail_log)
arima_retail |> 
  summary()
